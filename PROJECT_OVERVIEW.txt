    ports:
      - "5001:5000"                    # HTTP-Port
    depends_on:
      db:
        condition: service_healthy     # Warte auf DB
    volumes:
      - ./app:/app/app:ro              # Read-only App-Code

      - no-new-privileges:true         # Sicherheit
    user: "1000:1000"                  # Non-root User
